    for dep in all_deps:
        print(f"  {dep['source']} → {dep['target']}")
    
    store.close()

def example_1_topological_sort():
    """最简单的拓扑排序"""
    from topological_sorter import TopologicalSorter
    
    # 创建排序器
    sorter = TopologicalSorter()
    
    # 添加依赖关系：A → B → C
    sorter.add_edge("A", "B")  # A 依赖 B
    sorter.add_edge("B", "C")  # B 依赖 C
    
    # 执行排序
    sorted_ids, has_cycle = sorter.sort(all_ids=["A", "B", "C"])
    
    print(f"✅ 排序成功")
    print(f"顺序: {sorted_ids}")  # [C, B, A]
    print(f"有循环: {has_cycle}")  # False

# 2. 使用RAG检索相似用例
def example_2_rag_retrieval():
    """理解RAG检索机制"""
    from sentence_transformers import util
    import numpy as np
    
    # 模拟向量存储
    embeddings = {
        "login": np.array([0.5, 0.3, 0.8, 0.1]),
        "update_user": np.array([0.6, 0.4, 0.7, 0.2]),
    }
    
    # 新接口的向量（假设）
    new_interface = np.array([0.55, 0.35, 0.75, 0.15])
    
    # 计算相似度
    similarities = {}
    for name, vec in embeddings.items():
        sim = np.dot(new_interface, vec) / (np.linalg.norm(new_interface) * np.linalg.norm(vec))
        similarities[name] = sim
    
    # 排序
    sorted_items = sorted(similarities.items(), key=lambda x: x[1], reverse=True)
    
    print("✅ 相似用例排序:")
    for name, score in sorted_items:
        print(f"  {name}: {score:.3f}")

example_2_rag_retrieval()
```

# 1. 理解占位符替换机制
def example_1_placeholder_replacement():
    """理解占位符替换"""
    
    # 原始请求配置（包含占位符）
    config = {
        "headers": {"Authorization": "Bearer {token}"},
        "json": {"userId": "{userId}"}
    }
    
    # 从前一个测试提取的数据
    extracted_data = {
        "token": "abc123xyz",
        "userId": 42
    }
    
    # 替换逻辑
    def replace_placeholders(config, extracted_data):
        import json
        config_str = json.dumps(config)
        
        for key, value in extracted_data.items():
            config_str = config_str.replace(f"{{{key}}}", str(value))
        
        return json.loads(config_str)
    
    result = replace_placeholders(config, extracted_data)
    print("✅ 替换结果:")
    print(f"  Authorization: {result['headers']['Authorization']}")
    print(f"  userId: {result['json']['userId']}")

example_1_placeholder_replacement()

# 2. 理解嵌套字段提取
def example_2_nested_field_extraction():
    """理解嵌套字段提取"""
    
    # 响应内容
    response = {
        "code": 0,
        "data": {
            "token": "abc123",
            "user": {
                "id": 42,
                "name": "张三"
            }
        }
    }
    
    # 字段提取逻辑
    def extract_nested_field(obj, path):
        """提取嵌套字段 如 'data.user.id'"""
        parts = path.split(".")
        current = obj
        for part in parts:
            current = current.get(part, {})
        return current
    
    # 提取多个字段
    fields_to_extract = ["data.token", "data.user.id", "data.user.name"]
    extracted = {}
    for field in fields_to_extract:
        extracted[field] = extract_nested_field(response, field)
    
    print("✅ 提取结果:")
    for key, value in extracted.items():
        print(f"  {key}: {value}")

example_2_nested_field_extraction()
```

**代码位置**：
- `TestExecutor` 类 - 行846~1000
- `replace_placeholders` 方法 - 行860~880
- `extract_response_fields` 方法 - 行900~920
- 完整例子 - 行1002~1100

## 常见问题

### Q: 我没有Neo4j怎么办？

**A:** Neo4j是可选的，前3个学习模块（文档解析、Few-shot、占位符替换）不需要Neo4j。

建议：
1. 先学习其他模块
2. 使用Docker快速启动Neo4j（一条命令）
3. 或在云上使用Neo4j Sandbox（免费）

### Q: 我没有DeepSeek API密钥怎么办？

**A:** 大部分代码示例不需要真实API调用。

建议：
1. 运行本地的例子（只需理解概念）
2. 申请DeepSeek API（免费额度）
3. 替换为本地LLM（如Ollama）

### Q: 代码运行出错怎么办？

**A:** 常见错误：

| 错误 | 原因 | 解决方案 |
|------|------|---------|
| ModuleNotFoundError | 缺少依赖 | `pip install httpx requests ...` |
| ConnectionError | 无法连接Neo4j | 检查Neo4j是否启动 |
| TimeoutError | API超时 | 检查网络，增加超时时间 |
| JSONDecodeError | LLM返回格式错误 | 更新提示词 |

### Q: 能修改代码吗？

**A:** 可以！建议修改：

1. 系统提示词（看效果是否改善）
2. Few-shot示例（添加新的参考例子）
3. 占位符规则（支持更复杂的替换）
4. 字段提取规则（支持更多的字段格式）
