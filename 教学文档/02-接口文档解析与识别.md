## 文档解析智能体：3大步骤详解

这是智能体1的完整工作流程。我们用"大白话"把它拆解成3个清晰的步骤：

### 步骤1️⃣ 识别 → 提取内容

**目标**：从文件中提取原始内容，转换为文本/结构化数据

**工作方式**：
```
识别文件格式  
  ↓
  ├─ .pdf     → 用 pdfplumber 提取文本
  ├─ .docx    → 用 python-docx 读取 Word XML
  ├─ .xlsx    → 用 openpyxl 读取 Excel
  ├─ .json    → 直接 json.loads()
  ├─ .yaml    → 用 yaml.safe_load()
  ├─ .txt     → 按行读取
  └─ 其他     → 尝试通用方式
  
每种格式都有对应的专业库，确保解析准确性
```

**核心代码位置**：
```python
# backend/app/services/enhanced_document_parser.py
async def parse(file_path, file_type):
    # 1. 根据文件类型选择解析器
    # 2. 调用对应的 _parse_pdf_library / _parse_docx 等
    # 3. 返回原始提取的内容
```

**输出样例**：
```python
# 从PDF提取后
{
  "text": "API文档内容...",
  "tables": [
    {"headers": [...], "rows": [...]},
    ...
  ]
}

# 从JSON提取后  
{
  "apis": [
    {"name": "登录", "method": "POST", ...},
    ...
  ]
}
```

### 步骤2️⃣ 理解 → LLM标准化

**目标**：用 DeepSeek LLM 理解文档含义，统一成标准格式

**为什么需要LLM？**
```
问题场景：
- PDF: 可能有复杂的表格、混乱的格式
- Word: 可能有多种排版、手写标注  
- 文本: 可能是非结构化的自然语言说明

LLM的优势：
- 能理解自然语言
- 能识别表格中哪列是"参数名"哪列是"类型"
- 能补充缺失的信息（如识别"token"需要用在headers中）
```

#### 完整的Python代码示例

```python
import json
from typing import Dict, Any, List
import httpx

class LLMDocumentStandardizer:
    """使用LLM标准化文档内容"""
    
    def __init__(self, api_key: str, base_url: str = "https://api.deepseek.com/v1"):
        self.api_key = api_key
        self.base_url = base_url
        self.client = httpx.AsyncClient()
    
    async def standardize_document(self, extracted_content: str, file_type: str) -> Dict[str, Any]:
        """
        使用LLM标准化文档
        
        Args:
            extracted_content: 从文件提取的原始内容
            file_type: 文件类型 (pdf, word, json等)
        
        Returns:
            标准化的接口列表
        """
        # 构造系统提示词
        system_prompt = self._build_system_prompt(file_type)
        
        # 构造用户提示词
        user_prompt = f"""
请分析以下API文档内容，提取所有接口信息并转换为标准的JSON格式。

【原始文档内容】
{extracted_content}

【输出要求】
1. 提取所有接口
2. 转换为标准JSON格式
3. 补充缺失的重要信息
4. 识别参数的类型和必需性

请返回JSON格式的结果，不要添加其他说明。
"""
        
        # 调用DeepSeek API
        response = await self.client.post(
            f"{self.base_url}/chat/completions",
            headers={
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json"
            },
            json={
                "model": "deepseek-chat",
                "messages": [
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": user_prompt}
                ],
                "temperature": 0.2  # 降低温度使结果更确定
            }
        )
        
        result = response.json()
        llm_output = result["choices"][0]["message"]["content"]
        
        # 解析LLM输出
        standardized = self._parse_llm_output(llm_output)
        return standardized
    
    def _build_system_prompt(self, file_type: str) -> str:
        """构造系统提示词"""
        return f"""
你是一个专业的API文档分析师。
你的任务是从用户提供的API文档中提取接口信息，并转换为标准的JSON格式。

【标准JSON格式】
{{
    "interfaces": [
        {{
            "name": "接口名称",
            "method": "HTTP方法(GET/POST/PUT/DELETE)",
            "url": "接口路径或完整URL",
            "description": "接口描述说明",
            "request_headers": {{
                "Content-Type": "application/json",
                "其他必需请求头": "说明"
            }},
            "request_body": {{
                "type": "object",
                "required": ["字段1", "字段2"],
                "properties": {{
                    "字段名": {{
                        "type": "string|integer|boolean|object|array",
                        "description": "字段说明"
                    }}
                }}
            }},
            "response_body": {{
                "type": "object",
                "properties": {{
                    "code": {{"type": "integer", "description": "状态码"}},
                    "message": {{"type": "string", "description": "说明信息"}},
                    "data": {{
                        "type": "object",
                        "description": "返回数据",
                        "properties": {{}}
                    }}
                }}
            }},
            "status_code": 200,
            "examples": {{
                "request": {{}},
                "response": {{}}
            }}
        }}
    ]
}}

【注意事项】
1. 一定返回有效的JSON格式
2. 如果文档中没有某个字段，可以根据接口语义推断
3. 识别是否需要认证（Authorization头）
4. 列出所有可能的HTTP状态码
5. 标注哪些字段是必需的（required）

【文件类型】
你处理的是: {file_type} 文件

请按照上述标准格式返回结果。
"""
    
    def _parse_llm_output(self, output: str) -> Dict[str, Any]:
        """
        解析LLM输出
        """
        try:
            # 尝试直接解析JSON
            return json.loads(output)
        except json.JSONDecodeError:
            # 如果失败，尝试从markdown代码块中提取
            if "```json" in output:
                start = output.find("```json") + 7
                end = output.find("```", start)
                json_str = output[start:end].strip()
                return json.loads(json_str)
            elif "```" in output:
                start = output.find("```") + 3
                end = output.find("```", start)
                json_str = output[start:end].strip()
                return json.loads(json_str)
            else:
                # 最后尝试：查找 { 和 } 之间的内容
                start = output.find("{")
                end = output.rfind("}") + 1
                if start >= 0 and end > start:
                    json_str = output[start:end]
                    return json.loads(json_str)
            
            raise ValueError(f"无法解析LLM输出: {output}")


class DocumentParser:
    """完整的文档解析器"""
    
    def __init__(self, llm_standardizer: LLMDocumentStandardizer):
        self.llm_standardizer = llm_standardizer
    
    async def parse_pdf(self, file_path: str) -> Dict[str, Any]:
        """解析PDF文件"""
        import pdfplumber
        
        print(f"📄 解析PDF文件: {file_path}")
        
        content = ""
        with pdfplumber.open(file_path) as pdf:
            for page in pdf.pages:
                content += page.extract_text() + "\n"
        
        return content
    
    async def parse_word(self, file_path: str) -> Dict[str, Any]:
        """解析Word文件"""
        from docx import Document
        
        print(f"📄 解析Word文件: {file_path}")
        
        doc = Document(file_path)
        content = ""
        
        for para in doc.paragraphs:
            content += para.text + "\n"
        
        for table in doc.tables:
            for row in table.rows:
                content += " | ".join([cell.text for cell in row.cells]) + "\n"
        
        return content
    
    async def parse_json(self, file_path: str) -> Dict[str, Any]:
        """解析JSON文件"""
        print(f"📄 解析JSON文件: {file_path}")
        
        with open(file_path, 'r', encoding='utf-8') as f:
            return json.load(f)
    
    async def parse_document(self, file_path: str) -> Dict[str, Any]:
        """
        解析文档
        
        完整流程：
        1️⃣ 识别文件类型
        2️⃣ 提取原始内容  
        3️⃣ LLM标准化
        """
        import os
        
        # 识别文件类型
        _, ext = os.path.splitext(file_path)
        file_type = ext.lower().lstrip('.')
        
        print(f"🔍 识别文件格式: {file_type}")
        
        # 提取原始内容
        if file_type == 'pdf':
            extracted_content = await self.parse_pdf(file_path)
        elif file_type == 'docx':
            extracted_content = await self.parse_word(file_path)
        elif file_type == 'json':
            extracted_content = json.dumps(await self.parse_json(file_path), 
                                           ensure_ascii=False, indent=2)
        else:
            with open(file_path, 'r', encoding='utf-8') as f:
                extracted_content = f.read()
        
        print(f"✅ 提取完成，内容长度: {len(extracted_content)} 字符")
        
        # LLM标准化
        print(f"🤖 使用LLM标准化...")
        standardized = await self.llm_standardizer.standardize_document(
            extracted_content,
            file_type
        )
        
        print(f"✅ 标准化完成，获得 {len(standardized.get('interfaces', []))} 个接口")
        
        return standardized


# ===== 使用示例 =====

### 1. **OpenAPI/Swagger格式** (推荐)
这是**最结构化**的格式，包含完整的API定义

**特点**：
- 格式标准明确
- 字段定义完整
- 易于自动化解析
- 直接支持3.0和2.0版本

**示例结构**：
```yaml
openapi: 3.0.0
info:
  title: User API
  version: 1.0.0

paths:
  /users:
    get:
      summary: 获取用户列表
      parameters:
        - name: page
          in: query
          schema:
            type: integer
      responses:
        '200':
          description: 成功
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/UserList'

  /users/{id}:
    get:
      summary: 获取单个用户
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: integer
```

**解析重点**：
- `paths` → 接口URL
- `get/post/put/delete` → HTTP方法
- `parameters` → 请求参数
- `requestBody` → 请求体
- `responses` → 响应格式

## 接口识别逻辑

### 接口识别的关键点

#### 1. **URL识别**
```
目标：从文档中找到所有API的访问路径

识别模式：
- /api/xxx      (最常见)
- /v1/xxx       (带版本)
- /service/xxx  (带服务名)
- http://xxx    (完整URL)

检验：
- 必须包含至少一个 /
- 不能包含特殊字符（除-._~:/?#[]@!$&'()*+,;=）
```

#### 2. **HTTP方法识别**
```
直接查找文档中出现的：
- GET     (获取资源，无副作用)
- POST    (创建资源)
- PUT     (替换资源)
- DELETE  (删除资源)
- PATCH   (部分更新)
- HEAD    (类似GET但无响应体)
- OPTIONS (获取资源支持的方法)
```

#### 3. **参数识别**
```
位置分类：
┌─────────────────────────────────────┐
│ 1. Query Params (URL查询参数)       │
│    URL: /api/users?page=1&limit=10  │
│    识别：在?后的key=value            │
│                                      │
│ 2. Path Params (URL路径参数)        │
│    URL: /api/users/{id}             │
│    识别：{} 或 :param 标记           │
│                                      │
│ 3. Headers (请求头)                 │
│    识别：Content-Type, Authorization│
│           等HTTP标准请求头           │
│                                      │
│ 4. Body (请求体)                    │
│    POST/PUT/PATCH的请求体           │
│    识别：JSON/XML/FormData格式      │
└─────────────────────────────────────┘
```

#### 4. **响应识别**
```
响应包含：
- Status Code (HTTP状态码)
  常见：200(成功), 400(参数错误), 401(未授权)
        403(禁止), 404(未找到), 500(服务器错误)

- Response Headers
  常见：Content-Type, Set-Cookie等

- Response Body
  格式：通常是JSON
  识别：{}表示对象，[]表示数组
```

## 数据标准化

### 统一数据格式 (DocumentAPIInterface)

所有格式的文档都会被转换为以下统一格式：

```python
{
  "document_id": 123,          # 来源文档ID
  "project_id": 456,           # 项目ID
  
  # 接口基本信息
  "name": "接口名称",
  "method": "GET/POST/PUT/DELETE",
  "url": "/api/xxx",          # 完整URL或相对路径
  "service": "service.method", # 业务模块标识
  "description": "接口描述",
  
  # 请求信息
  "headers": {                 # JSON对象
    "Content-Type": "application/json",
    "Authorization": "Bearer token"
  },
  "params": {                  # 查询参数 (JSON)
    "page": {"type": "integer", "required": false},
    "limit": {"type": "integer", "required": false}
  },
  "request_body": {            # 请求体 (JSON Schema)
    "type": "object",
    "required": ["username"],
    "properties": {
      "username": {"type": "string"}
    }
  },
  
  # 响应信息
  "response_body": {           # 响应体示例 (JSON)
    "code": 0,
    "message": "success",
    "data": {}
  },
  "response_schema": {         # 响应Schema (JSON Schema)
    "type": "object",
    "properties": {
      "code": {"type": "integer"},
      "message": {"type": "string"}
    }
  },
  "status_code": 200,
  
  # 其他属性
  "tags": ["user", "auth"],
  "version": "1.0",
  "deprecated": false,
  "xjid": "30110"             # 项目特定字段
}
```

### 标准化的优势

✅ **一致性**：所有接口数据格式统一，便于后续处理
✅ **可查询性**：数据库查询时不需要关心原始格式
✅ **可复用**：同一个接口可用于多个场景（测试、文档等）
✅ **易维护**：修改一个接口定义时，所有地方都更新

## 核心代码示例

### DocumentParser的核心方法

```python
class DocumentParser:
    async def parse(self, file_path: str, file_type: str):
        """解析文档"""
        # 1. 选择解析器
        parser_map = {
            'json': self._parse_json,
            'yaml': self._parse_yaml,
            'pdf': self._parse_pdf,
            'docx': self._parse_docx,
            # ...
        }
        
        parser = parser_map.get(file_type)
        if not parser:
            raise ValueError(f"不支持格式: {file_type}")
        
        # 2. 提取原始数据
        raw_data = await parser(file_path)
        
        # 3. 标准化
        standardized = await self._standardize_data(raw_data, file_type)
        
        # 4. 验证
        self._validate_interfaces(standardized)
        
        return standardized
    
    async def _standardize_data(self, raw_data: dict, file_type: str):
        """数据标准化"""
        # 如果是OpenAPI，转换为标准格式
        if file_type == 'openapi':
            return self._convert_openapi_to_standard(raw_data)
        
        # 如果是其他格式，使用LLM理解
        if need_llm_enhance:
            prompt = f"""
            将以下文档内容转换为标准API接口定义：
            {raw_data}
            
            请返回JSON格式的接口列表
            """
            result = await self.llm_service.chat(prompt)
            return json.loads(result)
        
        return raw_data
```

## 最佳实践

### ✅ 推荐做法

1. **优先使用OpenAPI格式**
   - 最结构化
   - 自动化解析率最高
   - 易于维护

2. **文档中要包含完整信息**
   ```
   接口名称 ✓
   HTTP方法 ✓
   URL路径  ✓
   请求参数及类型 ✓
   响应格式及示例 ✓
   错误处理说明 ✓
   认证要求 ✓
   ```

3. **参数要明确标注**
   ```
   参数名 | 类型 | 必需 | 说明 | 示例
   ```

4. **提供响应示例**
   ```json
   // 成功响应(200)
   {
     "code": 0,
     "data": {...}
   }
   
   // 失败响应(400)
   {
     "code": -1,
     "message": "参数错误"
   }
   ```

### ❌ 避免的做法

1. ❌ 只有接口说明没有具体格式定义
2. ❌ 参数或响应字段标注不清楚
3. ❌ 混合多个版本的接口定义
4. ❌ 特殊字符导致格式混乱
5. ❌ 缺少错误响应说明

## 常见问题

**Q: 文档无法正确解析怎么办？**
A: 检查文件格式是否符合标准。对于非结构化文档（PDF/扫描件），使用LLM增强模式。

**Q: 同一接口在多个文档中出现怎么处理？**
A: 系统会进行去重，保留最新版本。建议维护单一接口文档源。

**Q: 如何处理不支持的格式？**
A: 可以先转换为JSON或YAML格式，或上传文本说明让LLM理解。
