```python
from typing import Dict, List, Any, Optional
from dataclasses import dataclass
from enum import Enum
import json
from datetime import datetime, timedelta
import logging

logger = logging.getLogger(__name__)

class DataType(Enum):
    """数据类型枚举"""
    STRING = "string"
    INTEGER = "integer"
    BOOLEAN = "boolean"
    ARRAY = "array"
    OBJECT = "object"
    EMAIL = "email"
    URL = "url"
    UUID = "uuid"
    PHONE = "phone"
    DATE = "date"
    DATETIME = "datetime"
    ENUM = "enum"

@dataclass
class FieldSchema:
    """字段模式定义"""
    name: str
    type: DataType
    required: bool = True
    default: Any = None
    enum_values: List[Any] = None
    pattern: str = None
    min_value: float = None
    max_value: float = None
    description: str = ""

class TestDataGenerator:
    """测试数据生成器"""
    
    def __init__(self):
        self.generated_ids = {}  # 用于追踪生成的ID
    
    def generate_value(self, field: FieldSchema) -> Any:
        """根据字段定义生成测试数据"""
        
        # 优先使用默认值
        if field.default is not None:
            return field.default
        
        # 根据类型生成
        if field.type == DataType.STRING:
            return self._generate_string(field)
        elif field.type == DataType.INTEGER:
            return self._generate_integer(field)
        elif field.type == DataType.BOOLEAN:
            return True
        elif field.type == DataType.EMAIL:
            return f"test_{self._random_id()}@example.com"
        elif field.type == DataType.UUID:
            import uuid
            return str(uuid.uuid4())
        elif field.type == DataType.PHONE:
            return f"1{self._random_number(10)}"
        elif field.type == DataType.DATE:
            return datetime.now().date().isoformat()
        elif field.type == DataType.DATETIME:
            return datetime.now().isoformat()
        elif field.type == DataType.ENUM:
            return field.enum_values[0] if field.enum_values else None
        elif field.type == DataType.ARRAY:
            return [self.generate_value(field)]
        elif field.type == DataType.OBJECT:
            return {}
        
        return None
    
    def _generate_string(self, field: FieldSchema) -> str:
        """生成字符串"""
        if field.pattern:
            return self._generate_by_pattern(field.pattern)
        return f"test_{self._random_id()}"
    
    def _generate_integer(self, field: FieldSchema) -> int:
        """生成整数"""
        import random
        min_val = int(field.min_value) if field.min_value else 1
        max_val = int(field.max_value) if field.max_value else 1000
        return random.randint(min_val, max_val)
    
    def _random_id(self) -> str:
        """生成随机ID"""
        import random
        return str(random.randint(10000, 99999))
    
    def _random_number(self, length: int) -> str:
        """生成随机数字"""
        import random
        return ''.join(str(random.randint(0, 9)) for _ in range(length))
    
    def _generate_by_pattern(self, pattern: str) -> str:
        """根据正则模式生成字符串"""
        from rstr import xeger
        return xeger(pattern)

class TestDataPreparer:
    """测试数据准备器"""
    
    def __init__(self, db_session, llm_service):
        self.db = db_session
        self.llm = llm_service
        self.data_generator = TestDataGenerator()
        self.dependency_resolver = DependencyResolver(db_session)
    
    async def prepare_test_data(
        self,
        interface: DocumentAPIInterface,
        dependencies: List[str],
        test_context: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        准备测试数据的完整流程
        
        Args:
            interface: 要测试的接口
            dependencies: 依赖的前置接口
            test_context: 测试上下文（共享的执行状态）
        
        Returns:
            准备好的测试数据
        """
        
        logger.info(f"开始准备测试数据: {interface.name}")
        
        # 步骤1：分析需要的数据字段
        required_fields = self._analyze_fields(interface)
        
        # 步骤2：生成/收集测试数据
        test_data = await self._collect_test_data(
            required_fields,
            interface,
            test_context
        )
        
        # 步骤3：处理依赖数据（从前置接口的响应中获取）
        if dependencies:
            dependency_data = await self._resolve_dependencies(
                dependencies,
                test_context
            )
            test_data.update(dependency_data)
        
        # 步骤4：占位符替换
        prepared_data = self._replace_placeholders(test_data, test_context)
        
        logger.info(f"✅ 数据准备完成，包含{len(prepared_data)}个字段")
        
        return prepared_data
    
    def _analyze_fields(self, interface: DocumentAPIInterface) -> List[FieldSchema]:
        """分析接口需要的数据字段"""
        
        fields = []
        
        # 从请求参数分析
        if interface.request_params:
            for param_name, param_info in interface.request_params.items():
                field = FieldSchema(
                    name=param_name,
                    type=self._map_type(param_info.get('type')),
                    required=param_info.get('required', True),
                    description=param_info.get('description', '')
                )
                fields.append(field)
        
        return fields
    
    def _map_type(self, type_name: str) -> DataType:
        """将类型名称映射到DataType"""
        type_map = {
            'string': DataType.STRING,
            'integer': DataType.INTEGER,
            'number': DataType.INTEGER,
            'boolean': DataType.BOOLEAN,
            'array': DataType.ARRAY,
            'object': DataType.OBJECT,
            'email': DataType.EMAIL,
            'url': DataType.URL,
            'uuid': DataType.UUID,
            'phone': DataType.PHONE,
            'date': DataType.DATE,
        }
        return type_map.get(type_name, DataType.STRING)
    
    async def _collect_test_data(
        self,
        fields: List[FieldSchema],
        interface: DocumentAPIInterface,
        test_context: Dict[str, Any]
    ) -> Dict[str, Any]:
        """收集/生成测试数据"""
        
        test_data = {}
        
        for field in fields:
            # 优先从上下文中查找数据
            if field.name in test_context:
                test_data[field.name] = test_context[field.name]
            # 其次使用LLM生成（对复杂字段）
            elif field.type in [DataType.OBJECT, DataType.ARRAY]:
                value = await self._generate_with_llm(field, interface)
                test_data[field.name] = value
            # 最后使用规则生成
            else:
                value = self.data_generator.generate_value(field)
                test_data[field.name] = value
        
        return test_data
    
    async def _generate_with_llm(
        self,
        field: FieldSchema,
        interface: DocumentAPIInterface
    ) -> Any:
        """使用LLM生成复杂字段"""
        
        prompt = f"""
        为接口 {interface.name} 生成一个 {field.name} 字段的测试数据。
        
        字段描述: {field.description}
        字段类型: {field.type.value}
        
        返回格式：只返回JSON格式的数据，不要其他说明。
        """
        
        response = await self.llm.chat(prompt, temperature=0.7)
        
        try:
            return json.loads(response)
        except:
            return {}
```

## 测试执行引擎

### TestExecutor主类

```python
from dataclasses import dataclass
from enum import Enum
import asyncio
import time
from http import HTTPStatus

class ExecutionStatus(Enum):
    """执行状态"""
    PENDING = "pending"
    RUNNING = "running"
    SUCCESS = "success"
    FAILED = "failed"
    SKIPPED = "skipped"

@dataclass
class TestExecutionResult:
    """测试执行结果"""
    interface_id: str
    interface_name: str
    status: ExecutionStatus
    request_data: Dict[str, Any]
    response_data: Dict[str, Any]
    assertions_passed: int
    assertions_failed: int
    execution_time: float
    error_message: Optional[str] = None
    response_headers: Dict = None
    response_status_code: int = None

class TestExecutor:
    """测试执行引擎"""
    
    def __init__(self, config: Dict = None):
        self.config = config or {}
        self.base_url = self.config.get('base_url', 'http://localhost:8000')
        self.timeout = self.config.get('timeout', 30)
        self.max_retries = self.config.get('max_retries', 3)
        self.retry_delay = self.config.get('retry_delay', 1)
        
        # 执行上下文（用于共享数据）
        self.context = {}
    
    async def execute_test(
        self,
        interface: DocumentAPIInterface,
        test_data: Dict[str, Any],
        assertions: List[str],
        dependencies: List[str] = None,
        test_context: Dict[str, Any] = None
    ) -> TestExecutionResult:
        """
        执行单个接口的测试
        
        Args:
            interface: 要测试的接口
            test_data: 测试数据
            assertions: 断言列表
            dependencies: 前置依赖
            test_context: 测试上下文
        
        Returns:
            测试执行结果
        """
        
        start_time = time.time()
        self.context = test_context or {}
        
        logger.info(f"开始执行: {interface.name}")
        
        try:
            # 步骤1：处理前置依赖
            if dependencies:
                await self._resolve_dependencies(
                    dependencies,
                    interface
                )
            
            # 步骤2：替换占位符
            prepared_data = self._replace_placeholders(
                test_data,
                self.context
            )
            
            # 步骤3：执行HTTP请求
            response = await self._send_request(
                interface,
                prepared_data
            )
            
            # 步骤4：执行断言
            passed, failed = await self._run_assertions(
                interface,
                response,
                assertions
            )
            
            # 步骤5：更新上下文（用于后续接口）
            self._update_context(interface, response)
            
            execution_time = time.time() - start_time
            
            return TestExecutionResult(
                interface_id=interface.id,
                interface_name=interface.name,
                status=ExecutionStatus.SUCCESS if failed == 0 else ExecutionStatus.FAILED,
                request_data=prepared_data,
                response_data=response.json() if response else {},
                assertions_passed=passed,
                assertions_failed=failed,
                execution_time=execution_time,
                response_status_code=response.status_code if response else None,
                response_headers=dict(response.headers) if response else {}
            )
        
        except Exception as e:
            execution_time = time.time() - start_time
            
            logger.error(f"❌ 执行失败: {str(e)}")
            
            return TestExecutionResult(
                interface_id=interface.id,
                interface_name=interface.name,
                status=ExecutionStatus.FAILED,
                request_data=test_data,
                response_data={},
                assertions_passed=0,
                assertions_failed=len(assertions),
                execution_time=execution_time,
                error_message=str(e)
            )
    
    async def _send_request(
        self,
        interface: DocumentAPIInterface,
        data: Dict[str, Any]
    ) -> Optional[Any]:
        """发送HTTP请求，支持重试"""
        
        import aiohttp
        
        url = f"{self.base_url}{interface.url}"
        method = interface.method.upper()
        
        # 准备请求头
        headers = self._prepare_headers(interface)
        
        # 准备请求体或参数
        if method in ['GET', 'HEAD', 'DELETE']:
            kwargs = {'params': data}
        else:
            kwargs = {'json': data}
        
        # 重试逻辑
        for attempt in range(self.max_retries):
            try:
                async with aiohttp.ClientSession() as session:
                    async with session.request(
                        method,
                        url,
                        headers=headers,
                        timeout=aiohttp.ClientTimeout(total=self.timeout),
                        **kwargs
                    ) as resp:
                        # 读取响应
                        response_data = await resp.text()
                        
                        logger.info(
                            f"✅ {method} {url} → {resp.status}"
                        )
                        
                        # 返回响应对象
                        resp.json_data = json.loads(response_data)
                        resp.text_data = response_data
                        return resp
            
            except asyncio.TimeoutError:
                if attempt < self.max_retries - 1:
                    logger.warning(f"⚠️ 超时，准备重试 ({attempt + 1}/{self.max_retries})")
                    await asyncio.sleep(self.retry_delay)
                    continue
                raise
            
            except Exception as e:
                if attempt < self.max_retries - 1:
                    logger.warning(f"⚠️ 请求失败: {str(e)}，准备重试")
                    await asyncio.sleep(self.retry_delay)
                    continue
                raise
        
        return None
    
    def _prepare_headers(self, interface: DocumentAPIInterface) -> Dict[str, str]:
        """准备请求头"""
        
        headers = {
            'Content-Type': 'application/json',
            'Accept': 'application/json'
        }
        
        # 添加认证信息
        if interface.requires_auth:
            token = self.context.get('auth_token')
            if token:
                headers['Authorization'] = f'Bearer {token}'
        
        # 添加自定义请求头
        if interface.headers:
            headers.update(interface.headers)
        
        return headers
    
    async def _run_assertions(
        self,
        interface: DocumentAPIInterface,
        response: Any,
        assertions: List[str]
    ) -> tuple:
        """执行断言"""
        
        passed = 0
        failed = 0
        
        for assertion in assertions:
            try:
                # 评估断言表达式
                result = eval(
                    assertion,
                    {
                        'response': response,
                        'status_code': response.status_code,
                        'json': response.json_data,
                        'text': response.text_data,
                        'headers': dict(response.headers)
                    }
                )
                
                if result:
                    passed += 1
                    logger.info(f"  ✅ {assertion}")
                else:
                    failed += 1
                    logger.warning(f"  ❌ {assertion} → False")
            
            except Exception as e:
                failed += 1
                logger.error(f"  ❌ {assertion} → Error: {str(e)}")
        
        return passed, failed
    
    def _update_context(
        self,
        interface: DocumentAPIInterface,
        response: Any
    ):
        """更新测试上下文，用于后续接口"""
        
        if not response:
            return
        
        # 保存响应数据供后续接口使用
        response_data = response.json_data
        
        # 常见的上下文更新
        # 从响应中提取ID字段
        for id_field in ['id', 'userId', 'orderId', 'token']:
            if id_field in response_data.get('data', {}):
                self.context[id_field] = response_data['data'][id_field]
            elif id_field in response_data:
                self.context[id_field] = response_data[id_field]
        
        # 特别处理认证token
        if interface.name.lower().contains('login'):
            if 'token' in response_data.get('data', {}):
                self.context['auth_token'] = response_data['data']['token']
        
        logger.info(f"  📦 上下文更新: {list(self.context.keys())}")
```

## 依赖解决和前置处理

### DependencyResolver实现

```python
class DependencyResolver:
    """依赖解决器"""
    
    def __init__(self, db_session, executor: TestExecutor):
        self.db = db_session
        self.executor = executor
        self.resolved = set()  # 已解决的依赖
    
    async def resolve_dependencies(
        self,
        interface_id: str,
        all_interfaces: Dict[str, DocumentAPIInterface],
        test_context: Dict[str, Any]
    ) -> bool:
        """
        解决接口的前置依赖
        
        Returns:
            True: 所有依赖都已成功解决
            False: 有依赖解决失败
        """
        
        # 查询此接口的依赖
        dependencies = await self._get_dependencies(interface_id)
        
        if not dependencies:
            return True
        
        logger.info(f"解决依赖: {len(dependencies)} 个前置接口")
        
        for dep_interface_id in dependencies:
            # 避免重复执行
            if dep_interface_id in self.resolved:
                continue
            
            # 递归解决依赖的依赖
            if not await self.resolve_dependencies(
                dep_interface_id,
                all_interfaces,
                test_context
            ):
                return False
            
            # 执行前置接口
            dep_interface = all_interfaces.get(dep_interface_id)
            if not dep_interface:
                logger.error(f"❌ 找不到依赖接口: {dep_interface_id}")
                return False
            
            logger.info(f"  → 执行前置接口: {dep_interface.name}")
            
            # 准备前置接口的测试数据
            preparer = TestDataPreparer(self.db, None)
            dep_data = await preparer.prepare_test_data(
                dep_interface,
                [],
                test_context
            )
            
            # 执行前置接口
            result = await self.executor.execute_test(
                dep_interface,
                dep_data,
                [],  # 前置接口通常不需要复杂断言
                test_context=test_context
            )
            
            # 检查前置接口是否成功
            if result.status != ExecutionStatus.SUCCESS:
                logger.error(f"❌ 前置接口执行失败: {dep_interface.name}")
                return False
            
            # 标记为已解决
            self.resolved.add(dep_interface_id)
            logger.info(f"  ✅ 前置接口执行成功")
        
        return True
    
    async def _get_dependencies(self, interface_id: str) -> List[str]:
        """从数据库查询接口的依赖"""
        
        # 这里假设有一个InterfaceDependency表
        query = self.db.query(InterfaceDependency).filter(
            InterfaceDependency.interface_id == interface_id
        ).all()
        
        return [dep.depends_on_id for dep in query]
```

## 占位符替换机制

### PlaceholderResolver实现

```python
import re

class PlaceholderResolver:
    """占位符替换器"""
    
    # 占位符模式：${variable_name} 或 {id}
    PLACEHOLDER_PATTERN = r'\$\{([a-zA-Z_][a-zA-Z0-9_]*)\}|\{([a-zA-Z_][a-zA-Z0-9_]*)\}'
    
    def __init__(self):
        pass
    
    def replace_in_url(
        self,
        url: str,
        context: Dict[str, Any]
    ) -> str:
        """
        替换URL中的占位符
        
        例如：
        原始: /api/users/{userId}/posts
        上下文: {userId: 123}
        结果: /api/users/123/posts
        """
        
        def replacer(match):
            var_name = match.group(1) or match.group(2)
            value = context.get(var_name)
            
            if value is None:
                logger.warning(f"⚠️ 占位符{var_name}未在上下文中找到")
                return match.group(0)
            
            return str(value)
        
        return re.sub(self.PLACEHOLDER_PATTERN, replacer, url)
    
    def replace_in_data(
        self,
        data: Dict[str, Any],
        context: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        递归替换数据中的占位符
        """
        
        if isinstance(data, dict):
            return {
                key: self.replace_in_data(value, context)
                for key, value in data.items()
            }
        
        elif isinstance(data, list):
            return [
                self.replace_in_data(item, context)
                for item in data
            ]
        
        elif isinstance(data, str):
            return self.replace_in_url(data, context)
        
        else:
            return data
    
    def replace_in_assertions(
        self,
        assertions: List[str],
        context: Dict[str, Any]
    ) -> List[str]:
        """替换断言中的占位符"""
        
        replaced = []
        
        for assertion in assertions:
            # 将 ${var} 替换为 context['var']
            def replacer(match):
                var_name = match.group(1) or match.group(2)
                value = context.get(var_name)
                
                if value is None:
                    return match.group(0)
                
                if isinstance(value, str):
                    return f'"{value}"'
                else:
                    return str(value)
            
            replaced_assertion = re.sub(
                self.PLACEHOLDER_PATTERN,
                replacer,
                assertion
            )
            
            replaced.append(replaced_assertion)
        
        return replaced
```

```python
async def run_full_test_suite(project_id: str):
    """运行完整的测试套件"""
    
    # 初始化
    db_session = get_db_session()
    executor = TestExecutor(config={'base_url': 'http://localhost:8000'})
    preparer = TestDataPreparer(db_session, llm_service)
    resolver = DependencyResolver(db_session, executor)
    placeholder_resolver = PlaceholderResolver()
    
    # 从数据库获取所有接口
    interfaces = await db_session.query(DocumentAPIInterface).filter(
        DocumentAPIInterface.project_id == project_id
    ).all()
    
    # 按拓扑顺序排序（依赖优先）
    sorted_interfaces = await topological_sort(interfaces)
    
    # 测试上下文（共享执行数据）
    test_context = {}
    results = []
    
    logger.info(f"开始执行{len(sorted_interfaces)}个接口的测试")
    
    for interface in sorted_interfaces:
        logger.info(f"\n{'='*60}")
        logger.info(f"测试: {interface.name} ({interface.method} {interface.url})")
        logger.info(f"{'='*60}")
        
        try:
            # 步骤1：准备测试数据
            test_data = await preparer.prepare_test_data(
                interface,
                interface.dependencies,
                test_context
            )
            
            # 步骤2：替换URL中的占位符
            interface.url = placeholder_resolver.replace_in_url(
                interface.url,
                test_context
            )
            
            # 步骤3：替换测试数据中的占位符
            test_data = placeholder_resolver.replace_in_data(
                test_data,
                test_context
            )
            
            # 步骤4：获取断言
            assertions = await db_session.query(TestAssertion).filter(
                TestAssertion.interface_id == interface.id
            ).all()
            assertion_strings = [a.assertion_code for a in assertions]
            
            # 步骤5：替换断言中的占位符
            assertion_strings = placeholder_resolver.replace_in_assertions(
                assertion_strings,
                test_context
            )
            
            # 步骤6：执行测试
            result = await executor.execute_test(
                interface,
                test_data,
                assertion_strings,
                interface.dependencies,
                test_context
            )
            
            results.append(result)
            
            # 打印结果
            logger.info(f"状态: {result.status.value}")
            logger.info(f"断言: {result.assertions_passed} 通过, {result.assertions_failed} 失败")
            logger.info(f"耗时: {result.execution_time:.2f}s")
            
        except Exception as e:
            logger.error(f"❌ 执行异常: {str(e)}")
            results.append(
                TestExecutionResult(
                    interface_id=interface.id,
                    interface_name=interface.name,
                    status=ExecutionStatus.FAILED,
                    request_data={},
                    response_data={},
                    assertions_passed=0,
                    assertions_failed=1,
                    execution_time=0,
                    error_message=str(e)
                )
            )
    
    # 生成报告
    report = generate_test_report(results)
    logger.info(f"\n\n{'='*60}")
    logger.info("测试报告")
    logger.info(f"{'='*60}")
    logger.info(f"总计: {len(results)} 个接口")
    logger.info(f"成功: {len([r for r in results if r.status == ExecutionStatus.SUCCESS])}")
    logger.info(f"失败: {len([r for r in results if r.status == ExecutionStatus.FAILED])}")
    logger.info(f"总耗时: {sum(r.execution_time for r in results):.2f}s")
    
    return report
```

## 性能优化

### 并发执行

```python
async def run_tests_concurrently(
    interfaces: List[DocumentAPIInterface],
    executor: TestExecutor,
    max_workers: int = 5
) -> List[TestExecutionResult]:
    """并发执行多个接口的测试（需要无依赖）"""
    
    from concurrent.futures import Semaphore
    
    semaphore = asyncio.Semaphore(max_workers)
    
    async def execute_with_semaphore(interface):
        async with semaphore:
            return await executor.execute_test(interface, {}, [])
    
    tasks = [execute_with_semaphore(iface) for iface in interfaces]
    return await asyncio.gather(*tasks)
```
