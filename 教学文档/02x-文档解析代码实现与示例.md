## LLM文档标准化实现

### 设计思路

```
原始文档（PDF/Word/JSON等）
  ↓
【提取阶段】解析文件格式，获取原始文本内容
  ↓
【理解阶段】使用DeepSeek LLM理解内容，识别接口信息
  ↓
【标准化】转换为统一的JSON格式（DocumentAPIInterface）
  ↓
标准化后的接口定义
```

### 核心代码：LLMDocumentStandardizer类

```python
import json
from typing import Dict, Any, Optional
import httpx
import asyncio

class LLMDocumentStandardizer:
    """使用LLM标准化文档内容"""
    
    def __init__(self, api_key: str, base_url: str = "https://api.deepseek.com/v1"):
        """
        初始化LLM标准化器
        
        Args:
            api_key: DeepSeek API密钥
            base_url: API基础URL
        """
        self.api_key = api_key
        self.base_url = base_url
        self.client = httpx.AsyncClient(timeout=30.0)
    
    async def standardize_document(self, extracted_content: str, file_type: str) -> Dict[str, Any]:
        """
        使用LLM标准化文档
        
        Args:
            extracted_content: 从文件提取的原始内容（文本）
            file_type: 文件类型 (pdf, word, json, yaml等)
        
        Returns:
            标准化的接口列表字典
            {
                "interfaces": [
                    {
                        "name": "接口名称",
                        "method": "GET|POST|PUT|DELETE",
                        "url": "接口路径",
                        ...
                    }
                ]
            }
        
        Raises:
            ValueError: 如果LLM输出无法解析为JSON
            httpx.HTTPError: 如果API调用失败
        """
        # 步骤1: 构造系统提示词（定义LLM的角色和输出格式）
        system_prompt = self._build_system_prompt(file_type)
        
        # 步骤2: 构造用户提示词（实际的任务和数据）
        user_prompt = self._build_user_prompt(extracted_content)
        
        # 步骤3: 调用DeepSeek API
        try:
            response = await self.client.post(
                f"{self.base_url}/chat/completions",
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json"
                },
                json={
                    "model": "deepseek-chat",
                    "messages": [
                        {"role": "system", "content": system_prompt},
                        {"role": "user", "content": user_prompt}
                    ],
                    "temperature": 0.2,  # 降低温度使输出更确定、准确
                    "max_tokens": 4096
                }
            )
            
            result = response.json()
            if "error" in result:
                raise ValueError(f"API错误: {result['error']}")
            
            llm_output = result["choices"][0]["message"]["content"]
            
        except httpx.HTTPError as e:
            raise ValueError(f"API调用失败: {str(e)}")
        
        # 步骤4: 解析和验证LLM输出
        standardized = self._parse_llm_output(llm_output)
        
        return standardized
    
    def _build_system_prompt(self, file_type: str) -> str:
        """
        构造系统提示词
        
        系统提示词定义：
        - LLM的角色（API文档分析师）
        - 输出格式（标准JSON）
        - 字段含义和类型
        - 处理特殊情况的方法
        """
        return f"""你是一个专业的API文档分析师，擅长从各种格式的API文档中提取信息。

【您的任务】
从用户提供的API文档中提取所有接口信息，并转换为标准的JSON格式。

【标准输出格式】
{{
    "interfaces": [
        {{
            "name": "接口的名称或简要说明",
            "method": "HTTP方法(GET|POST|PUT|DELETE|PATCH|HEAD|OPTIONS)",
            "url": "接口的路径或完整URL，如 /api/users 或 /api/users/{{id}}",
            "description": "接口的功能说明，1-2句话",
            "request_headers": {{
                "Content-Type": "application/json",
                "Authorization": "Bearer {{token}}", 
                "其他必需的请求头": "说明"
            }},
            "request_body": {{
                "type": "object",
                "required": ["必需参数1", "必需参数2"],
                "properties": {{
                    "参数名": {{
                        "type": "string|integer|boolean|array|object",
                        "description": "参数说明",
                        "example": "参数示例值"
                    }}
                }}
            }},
            "response_body": {{
                "type": "object",
                "properties": {{
                    "code": {{"type": "integer", "description": "响应状态码，0表示成功"}},
                    "message": {{"type": "string", "description": "响应信息"}},
                    "data": {{
                        "type": "object",
                        "description": "响应数据",
                        "properties": {{}}
                    }}
                }}
            }},
            "status_code": 200,
            "examples": {{
                "request": {{}},
                "response": {{}}
            }},
            "auth_required": true,
            "rate_limit": "可选：速率限制说明"
        }}
    ]
}}

【关键规则】
1. 【JSON格式】：必须返回有效的JSON格式，不要其他文字
2. 【字段必需性】：标注哪些参数是必需的（required数组）
3. 【参数类型】：识别每个参数的数据类型
4. 【认证识别】：识别接口是否需要认证令牌
5. 【HTTP方法】：准确识别REST接口的HTTP方法
6. 【推理补充】：如果文档不完整，根据接口语义进行合理推断
7. 【多个接口】：如果文档中有多个接口，都要列在interfaces数组中

【您处理的文件类型】
{file_type}

请按照上述标准格式返回结果。"""
    
    def _build_user_prompt(self, content: str) -> str:
        """构造用户提示词"""
        max_chars = 3000  # 限制字符长度以避免超出token限制
        if len(content) > max_chars:
            content = content[:max_chars] + "\n... [内容已截断] ..."
        
        return f"""请分析以下API文档内容，提取所有接口信息并转换为标准的JSON格式。

【原始文档内容】
{content}

【提取要求】
1. 提取所有接口，每个接口都要有method、url、description
2. 识别request_body中的所有参数及其类型
3. 识别response_body的结构
4. 标注必需参数（required数组）
5. 识别是否需要认证（Authorization）

请返回有效的JSON格式，不要任何其他说明。"""
    
    def _parse_llm_output(self, output: str) -> Dict[str, Any]:
        """
        解析LLM输出
        
        尝试多种方式解析，提高容错能力
        """
        output = output.strip()
        
        # 尝试方式1：直接解析JSON
        try:
            return json.loads(output)
        except json.JSONDecodeError:
            pass
        
        # 尝试方式2：从```json代码块中提取
        if "```json" in output:
            try:
                start = output.find("```json") + 7
                end = output.find("```", start)
                if end > start:
                    json_str = output[start:end].strip()
                    return json.loads(json_str)
            except (json.JSONDecodeError, ValueError):
                pass
        
        # 尝试方式3：从```代码块中提取
        if "```" in output:
            try:
                parts = output.split("```")
                for i in range(1, len(parts), 2):
                    json_str = parts[i].strip()
                    if json_str.startswith("json"):
                        json_str = json_str[4:].strip()
                    try:
                        return json.loads(json_str)
                    except json.JSONDecodeError:
                        continue
            except:
                pass
        
        # 尝试方式4：查找{ }之间的内容
        try:
            start = output.find("{")
            end = output.rfind("}") + 1
            if start >= 0 and end > start:
                json_str = output[start:end]
                return json.loads(json_str)
        except json.JSONDecodeError:
            pass
        
        # 都失败了，抛出异常
        raise ValueError(
            f"无法解析LLM输出。输出内容：\n{output[:500]}"
        )

## 多格式文档解析器

### 核心代码：DocumentParser类

```python
import os
import json
from pathlib import Path
from typing import Dict, Any, Union
import asyncio

class DocumentParser:
    """
    完整的文档解析器
    
    支持多种文件格式：PDF、Word、JSON、YAML、文本
    """
    
    def __init__(self, llm_standardizer: LLMDocumentStandardizer):
        """
        初始化解析器
        
        Args:
            llm_standardizer: LLMDocumentStandardizer实例
        """
        self.llm_standardizer = llm_standardizer
    
    async def parse_pdf(self, file_path: str) -> str:
        """
        解析PDF文件
        
        Args:
            file_path: PDF文件路径
        
        Returns:
            PDF中提取的文本内容
        
        Requires: pdfplumber library
        """
        try:
            import pdfplumber
        except ImportError:
            raise ImportError("请先安装: pip install pdfplumber")
        
        print(f"📄 解析PDF文件: {file_path}")
        
        content = ""
        try:
            with pdfplumber.open(file_path) as pdf:
                for page_num, page in enumerate(pdf.pages, 1):
                    # 提取文本
                    text = page.extract_text()
                    if text:
                        content += f"\n--- 第{page_num}页 ---\n{text}"
                    
                    # 提取表格
                    tables = page.extract_tables()
                    if tables:
                        for table in tables:
                            content += "\n【表格】\n"
                            for row in table:
                                content += " | ".join([str(cell) if cell else "" for cell in row]) + "\n"
        except Exception as e:
            print(f"⚠️  PDF解析出错: {e}")
            raise
        
        return content
    
    async def parse_word(self, file_path: str) -> str:
        """
        解析Word文件
        
        Args:
            file_path: .docx文件路径
        
        Returns:
            Word文档中提取的内容
        
        Requires: python-docx library
        """
        try:
            from docx import Document
        except ImportError:
            raise ImportError("请先安装: pip install python-docx")
        
        print(f"📄 解析Word文件: {file_path}")
        
        content = ""
        try:
            doc = Document(file_path)
            
            # 提取段落
            for para in doc.paragraphs:
                if para.text.strip():
                    content += para.text + "\n"
            
            # 提取表格
            for table_num, table in enumerate(doc.tables, 1):
                content += f"\n【表格{table_num}】\n"
                for row in table.rows:
                    cells_text = [cell.text for cell in row.cells]
                    content += " | ".join(cells_text) + "\n"
        
        except Exception as e:
            print(f"⚠️  Word解析出错: {e}")
            raise
        
        return content
    
    async def parse_json(self, file_path: str) -> str:
        """
        解析JSON文件
        
        Args:
            file_path: JSON文件路径
        
        Returns:
            JSON内容的字符串表示
        """
        print(f"📄 解析JSON文件: {file_path}")
        
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            return json.dumps(data, ensure_ascii=False, indent=2)
        except Exception as e:
            print(f"⚠️  JSON解析出错: {e}")
            raise
    
    async def parse_yaml(self, file_path: str) -> str:
        """
        解析YAML文件
        
        Args:
            file_path: YAML文件路径
        
        Returns:
            YAML内容的字符串表示
        
        Requires: pyyaml library
        """
        try:
            import yaml
        except ImportError:
            raise ImportError("请先安装: pip install pyyaml")
        
        print(f"📄 解析YAML文件: {file_path}")
        
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                data = yaml.safe_load(f)
            return yaml.dump(data, allow_unicode=True, default_flow_style=False)
        except Exception as e:
            print(f"⚠️  YAML解析出错: {e}")
            raise
    
    async def parse_text(self, file_path: str) -> str:
        """解析纯文本文件"""
        print(f"📄 解析文本文件: {file_path}")
        
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                return f.read()
        except Exception as e:
            print(f"⚠️  文本解析出错: {e}")
            raise
    
    async def parse_document(self, file_path: str) -> Dict[str, Any]:
        """
        解析任何支持的文档格式
        
        完整3步流程：
        1️⃣ 识别文件类型
        2️⃣ 提取原始内容
        3️⃣ LLM标准化
        
        Args:
            file_path: 待解析的文件路径
        
        Returns:
            标准化的接口信息字典
        
        Raises:
            FileNotFoundError: 文件不存在
            ValueError: 不支持的文件格式
        """
        # 检查文件是否存在
        if not os.path.exists(file_path):
            raise FileNotFoundError(f"文件不存在: {file_path}")
        
        # 步骤1️⃣：识别文件类型
        _, ext = os.path.splitext(file_path)
        file_type = ext.lower().lstrip('.')
        
        print(f"\n🔍 识别文件格式: {file_type}")
        
        # 步骤2️⃣：提取原始内容
        try:
            if file_type in ['pdf']:
                extracted_content = await self.parse_pdf(file_path)
            elif file_type in ['docx', 'doc']:
                extracted_content = await self.parse_word(file_path)
            elif file_type in ['json']:
                extracted_content = await self.parse_json(file_path)
            elif file_type in ['yaml', 'yml']:
                extracted_content = await self.parse_yaml(file_path)
            else:
                # 默认作为纯文本处理
                extracted_content = await self.parse_text(file_path)
            
            print(f"✅ 提取完成，内容长度: {len(extracted_content)} 字符")
        
        except Exception as e:
            print(f"❌ 内容提取失败: {e}")
            raise
        
        # 步骤3️⃣：LLM标准化
        try:
            print(f"🤖 使用LLM标准化...")
            standardized = await self.llm_standardizer.standardize_document(
                extracted_content,
                file_type
            )
            
            interface_count = len(standardized.get('interfaces', []))
            print(f"✅ 标准化完成，获得 {interface_count} 个接口")
            
            return standardized
        
        except Exception as e:
            print(f"❌ LLM标准化失败: {e}")
            raise


# 使用示例

            print(f"  请求参数: {list(req_props.keys())}")

        resp_props = interface.get('response_body', {}).get('properties', {})
        if resp_props:
            print(f"  响应字段: {list(resp_props.keys())}")

## 常见问题和优化

### Q1: 如何处理很大的PDF文件？

**A:** 可以分页处理，避免一次性加载整个文件：

```python
async def parse_pdf_chunked(self, file_path: str, max_pages: int = 10) -> str:
    """分页处理PDF，避免超出token限制"""
    import pdfplumber
    
    content = ""
    with pdfplumber.open(file_path) as pdf:
        for page_num in range(min(max_pages, len(pdf.pages))):
            page = pdf.pages[page_num]
            text = page.extract_text()
            if text:
                content += f"\n--- 第{page_num+1}页 ---\n{text}"
    
    return content
```

### Q2: LLM返回的JSON格式不标准怎么办？

**A:** 已经内置了容错机制，但可以进一步优化：

```python
def _parse_llm_output_robust(self, output: str) -> Dict[str, Any]:
    """更强大的JSON解析器"""
    import re
    
    # 尝试多种JSON查找方式
    patterns = [
        r'\{[\s\S]*\}',  # 查找任何{}括号
        r'\[\s*\{[\s\S]*\}\s*\]',  # 查找数组
    ]
    
    for pattern in patterns:
        match = re.search(pattern, output)
        if match:
            try:
                return json.loads(match.group())
            except json.JSONDecodeError:
                continue
    
    raise ValueError("无法解析JSON")
```

### Q3: 如何加速处理多个文档？

**A:** 使用并发处理：

```python
async def parse_multiple_documents(self, file_paths: list) -> list:
    """并发解析多个文档"""
    tasks = [
        self.parse_document(file_path)
        for file_path in file_paths
    ]
    return await asyncio.gather(*tasks)
```
