## 多格式解析概览

### 为什么需要支持多种格式？

```
现实情况：
  - 部分企业只有 PDF 或 Word 文档
  - 不是所有项目都有标准的 OpenAPI 文档
  - 不同行业、不同阶段的文档格式各异
  
解决方案：
  系统支持多种格式的自动识别和解析
  用户可以上传任何格式的文档，我们来处理
```

**解析步骤**：

```python
import yaml
import json

def parse_openapi(file_path):
    """解析OpenAPI文件"""
    
    # 1. 读取文件
    if file_path.endswith('.json'):
        with open(file_path) as f:
            spec = json.load(f)
    else:  # .yaml 或 .yml
        with open(file_path) as f:
            spec = yaml.safe_load(f)
    
    interfaces = []
    
    # 2. 遍历所有路径
    for path, path_item in spec['paths'].items():
        # 3. 遍历所有HTTP方法
        for method, operation in path_item.items():
            if method not in ['get', 'post', 'put', 'delete', 'patch']:
                continue
            
            # 4. 提取接口信息
            interface = {
                'name': operation.get('summary', ''),
                'method': method.upper(),
                'url': path,
                'description': operation.get('description', ''),
                
                # 提取参数
                'params': extract_params(operation.get('parameters', [])),
                'request_body': extract_request_body(operation.get('requestBody')),
                
                # 提取响应
                'response_body': extract_response(operation.get('responses', {})),
                'tags': operation.get('tags', [])
            }
            
            interfaces.append(interface)
    
    return interfaces

def extract_params(parameters):
    """提取查询参数和路径参数"""
    params = {}
    for param in parameters:
        param_name = param['name']
        params[param_name] = {
            'type': param.get('schema', {}).get('type', 'string'),
            'required': param.get('required', False),
            'description': param.get('description', '')
        }
    return params

def extract_request_body(request_body):
    """提取请求体"""
    if not request_body:
        return None
    
    schema = request_body.get('content', {}).get('application/json', {}).get('schema', {})
    return schema

def extract_response(responses):
    """提取响应"""
    success_response = responses.get('200') or responses.get('201')
    if not success_response:
        return None
    
    schema = success_response.get('content', {}).get('application/json', {}).get('schema', {})
    return schema
```

### Swagger 2.0（兼容）

**特点**：
- ✓ 较老的标准格式
- ✓ 仍被广泛使用
- ✓ 解析相对容易
- ✓ 自动化解析率：90%+

**关键字段**：
- `swagger`: "2.0"
- `paths`: 接口定义
- `definitions`: 数据模型定义

**与OpenAPI 3.0的主要区别**：

| 特性 | Swagger 2.0 | OpenAPI 3.0 |
|------|------------|-----------|
| 参数定义 | `parameters` (数组) | `parameters` (数组) |
| 请求体 | `parameters` (body类型) | `requestBody` (对象) |
| 内容类型 | `consumes/produces` | `content` |
| 数据模型 | `definitions` | `components/schemas` |

**解析相同**，只需调整字段名称。

## 文档格式

### PDF 文档

**解析方法**：
1. **文本提取**（优先）：如果PDF是数字化的
2. **OCR识别**：如果PDF是扫描件
3. **LLM理解**：辅助识别接口信息

**代码示例**：

```python
import pdfplumber
from pdf2image import convert_from_path
import pytesseract

def parse_pdf(file_path):
    """解析PDF文档"""
    
    content = ""
    
    # 方法1: 尝试直接提取文本
    with pdfplumber.open(file_path) as pdf:
        for page in pdf.pages:
            text = page.extract_text()
            tables = page.extract_tables()
            
            if text:
                content += text + "\n"
            
            # 提取表格
            for table in tables or []:
                for row in table:
                    content += " | ".join(str(cell) for cell in row) + "\n"
    
    # 如果文本很少，可能是扫描件，使用OCR
    if len(content.strip()) < 100:
        # 方法2: 使用OCR识别扫描件
        images = convert_from_path(file_path)
        for image in images:
            text = pytesseract.image_to_string(image, lang='chi_sim+eng')
            content += text + "\n"
    
    return content
```

**识别接口的技巧**：

```python
import re

def extract_interfaces_from_text(text):
    """从PDF文本中识别接口"""
    
    interfaces = []
    
    # 正则模式：匹配常见的接口描述格式
    # 例如："[POST] /api/users" 或 "接口：POST /api/users"
    pattern = r'(POST|GET|PUT|DELETE|PATCH)\s+(/[\w/:\-_{}]*)'
    
    matches = re.finditer(pattern, text, re.IGNORECASE)
    for match in matches:
        method = match.group(1).upper()
        url = match.group(2)
        
        # 找出这个接口的上下文（前后几行）
        start = max(0, match.start() - 200)
        end = min(len(text), match.end() + 200)
        context = text[start:end]
        
        interfaces.append({
            'method': method,
            'url': url,
            'context': context  # 供LLM理解
        })
    
    return interfaces
```

### Word 文档 (.docx)

**解析方法**：使用`python-docx`库读取Word XML结构

**代码示例**：

```python
from docx import Document
from docx.table import Table

def parse_word_document(file_path):
    """解析Word文档"""
    
    doc = Document(file_path)
    interfaces = []
    current_api_info = {}
    
    # 遍历所有段落和表格
    for element in doc.element.body:
        if element.tag.endswith('p'):  # 段落
            # 解析段落内容
            paragraph_text = ''.join([t.text for t in element.findall('.//' + '{http://schemas.openxmlformats.org/wordprocessingml/2006/main}t')])
            
            # 识别接口信息
            if 'POST' in paragraph_text or 'GET' in paragraph_text:
                # 可能是接口定义
                lines = paragraph_text.split('\n')
                for line in lines:
                    if any(method in line for method in ['POST', 'GET', 'PUT', 'DELETE']):
                        current_api_info['definition'] = line
        
        elif element.tag.endswith('tbl'):  # 表格
            # 解析表格
            table = Table(element, doc)
            interfaces.extend(parse_word_table(table))
    
    return interfaces

def parse_word_table(table):
    """解析Word表格中的API定义"""
    
    interfaces = []
    headers = []
    
    # 第一行是表头
    header_cells = table.rows[0].cells
    for cell in header_cells:
        headers.append(cell.text)
    
    # 后续行是数据
    for row in table.rows[1:]:
        cells = row.cells
        interface_dict = {}
        
        for i, cell in enumerate(cells):
            if i < len(headers):
                interface_dict[headers[i]] = cell.text
        
        interfaces.append(interface_dict)
    
    return interfaces
```

                    pass

            if interface:
                interfaces.append(interface)

def parse_row_to_interface(cells):
    """将Excel行转换为接口定义"""

        'name': cells[1],
        'method': cells[2],

    }

**解析代码**：

```python
import re

def parse_markdown(file_path):
    """解析Markdown文档"""
    
    with open(file_path, encoding='utf-8') as f:
        content = f.read()
    
    interfaces = []
    
    # 按接口分割（每个接口通常是一个 ### 标题）
    sections = re.split(r'^###\s+(.+)$', content, flags=re.MULTILINE)
    
    for i in range(1, len(sections), 2):
        api_name = sections[i]
        api_content = sections[i + 1] if i + 1 < len(sections) else ""
        
        interface = parse_markdown_section(api_name, api_content)
        interfaces.append(interface)
    
    return interfaces

def parse_markdown_section(name, content):
    """解析单个接口的Markdown内容"""
    
    interface = {'name': name.strip()}
    
    # 提取请求方法
    method_match = re.search(r'\*\*请求方法\*?\*?:\s*(GET|POST|PUT|DELETE|PATCH)', content)
    if method_match:
        interface['method'] = method_match.group(1)
    
    # 提取URL
    url_match = re.search(r'\*\*URL\*?\*?:\s*(`?/?[\w/:-]+`?)', content)
    if url_match:
        interface['url'] = url_match.group(1).strip('`')
    
    # 提取参数表格
    table_match = re.search(r'\|\s*参数\s*\|.*?\n\|[\s\-|:]+\n((?:\|.*?\n)+)', content, re.DOTALL)
    if table_match:
        table_content = table_match.group(1)
        interface['params'] = parse_markdown_table(table_content)
    
    return interface

def parse_markdown_table(table_content):
    """解析Markdown表格"""
    params = []
    rows = table_content.strip().split('\n')
    
    for row in rows:
        cells = [cell.strip() for cell in row.split('|')]
        if len(cells) >= 2:
            params.append({
                'name': cells[1],
                'type': cells[2] if len(cells) > 2 else 'string',
                'required': cells[3] if len(cells) > 3 else 'no',
                'description': cells[4] if len(cells) > 4 else ''
            })
    
    return params
```

### TXT 纯文本

**解析策略**：使用正则表达式和关键词匹配

```python
def parse_txt(file_path):
    """解析纯文本文件"""
    
    with open(file_path, encoding='utf-8') as f:
        lines = f.readlines()
    
    interfaces = []
    current_api = {}
    
    for line in lines:
        line = line.strip()
        
        # 识别接口开始（关键词：接口、API、Method等）
        if any(keyword in line for keyword in ['接口:', 'API:', '方法:']):
            if current_api:
                interfaces.append(current_api)
            current_api = {'description': line}
        
        # 识别HTTP方法
        elif any(method in line for method in ['POST', 'GET', 'PUT', 'DELETE']):
            current_api['method'] = line.split()[0]
            if '/' in line:
                current_api['url'] = [part for part in line.split() if '/' in part][0]
    
    if current_api:
        interfaces.append(current_api)
    
    return interfaces
```

## 专用格式

### JMeter 脚本 (.jmx)

JMeter是性能测试工具，脚本是XML格式。我们可以从中提取HTTP请求的接口信息：

```python
import xml.etree.ElementTree as ET

def parse_jmeter_jmx(file_path):
    """从JMeter脚本中提取接口信息"""
    
    tree = ET.parse(file_path)
    root = tree.getroot()
    
    interfaces = []
    
    # JMeter中的HTTP请求是 HTTPSamplerProxy 元素
    for sampler in root.findall('.//HTTPSamplerProxy'):
        # 提取请求信息
        name = sampler.get('guiclass', '')
        
        # 获取子元素
        domain = None
        port = None
        path = None
        method = None
        
        for elem in sampler:
            if elem.tag == 'elementProp' and elem.get('name') == 'HTTPsampler.Arguments':
                # 参数列表
                pass
            elif hasattr(elem, 'text'):
                if 'domain' in elem.tag:
                    domain = elem.text
                elif 'port' in elem.tag:
                    port = elem.text
                elif 'path' in elem.tag:
                    path = elem.text
                elif 'method' in elem.tag:
                    method = elem.text
        
        if path and method:
            interfaces.append({
                'name': sampler.get('testname', ''),
                'method': method,
                'url': f"/{path}" if path else '/',
                'domain': domain
            })
    
    return interfaces
```
