```python
from typing import Dict, List, Optional, Any

@dataclass
class TestCaseGenerationRequest:

class TestCaseGenerator:

    def __init__(
        self,
        llm_service,
        vector_store,
        config: Dict = None
    ):
        """
        初始化生成器
        
        Args:
            llm_service: LLM服务（调用OpenAI/Claude等）
            vector_store: 向量存储（用于RAG检索）
            config: 配置参数
        """

    def _select_few_shot_examples(
        self,
        request: TestCaseGenerationRequest
    ) -> List[Dict]:
        """
        为当前接口选择最相关的Few-shot示例
        """
        
        # 优先级1：同样的HTTP方法和相似的URL
        candidates = []
        
        for example in self.few_shot_examples:
            if example['method'] == request.http_method:
                # 检查URL相似性
                if self._url_similarity(example['url'], request.url) > 0.5:
                    candidates.append(example)
        
        if candidates:
            return candidates[:self.few_shot_count]
        
        # 优先级2：同样的HTTP方法
        candidates = [
            e for e in self.few_shot_examples
            if e['method'] == request.http_method
        ]
        
        if candidates:
            return candidates[:self.few_shot_count]
        
        # 降级：返回所有示例
        return self.few_shot_examples[:self.few_shot_count]
    
    def _url_similarity(self, url1: str, url2: str) -> float:
        """计算两个URL的相似度"""
        # 简化版实现
        from difflib import SequenceMatcher
        return SequenceMatcher(None, url1, url2).ratio()


async def generate(
        self,
        request: TestCaseGenerationRequest
    ) -> TestCaseGenerationResult:
        """
        主生成流程：Few-shot + RAG + LLM
        """
        
        logger.info(f"开始生成测试用例: {request.interface_name}")
        
        try:
            # 步骤1：选择Few-shot示例
            few_shot_examples = self._select_few_shot_examples(request)
            
            # 步骤2：从RAG检索相似的测试用例
            rag_cases = await self._retrieve_similar_cases(request)
            
            # 步骤3：构造完整的提示词
            prompt = self._build_prompt(request, few_shot_examples, rag_cases)
            
            # 步骤4：调用LLM生成
            response = await self.llm_service.create(
                model=self.model,
                messages=[
                    {
                        "role": "system",
                        "content": self._get_system_prompt()
                    },
                    {
                        "role": "user",
                        "content": prompt
                    }
                ],
                temperature=self.temperature,
                max_tokens=self.max_tokens
            )
            
            test_code = response['choices'][0]['message']['content']
            
            # 步骤5：解析测试用例
            test_cases = self._parse_test_cases(test_code, request)
            
            # 步骤6：存储到RAG库（用于后续检索）
            await self._store_to_rag(request, test_code)
            
            result = TestCaseGenerationResult(
                interface_id=request.interface_id,
                test_code=test_code,
                test_cases=test_cases,
                generated_at=datetime.now(),
                model_used=self.model,
                tokens_used=response.get('usage', {}).get('total_tokens', 0),
                few_shot_count=len(few_shot_examples),
                rag_score=rag_cases[0]['score'] if rag_cases else None
            )
            
            logger.info(f"✅ 成功生成: {len(test_cases)} 个测试用例")
            return result
            
        except Exception as e:
            logger.error(f"❌ 生成失败: {str(e)}")
            raise
```

```python
def _get_system_prompt(self) -> str:
    """获取系统级提示词"""
    return """
你是一个资深的API测试工程师，拥有10年的测试经验。

你的工作是为给定的API接口生成高质量的Python Pytest测试用例。

重要规则：
1. 生成的代码必须是可以直接运行的Python代码
2. 必须包括正常情况和异常情况的测试
3. 必须进行充分的断言检查

- 参数验证测试
- 认证失败测试
- 权限不足测试
- 依赖接口缺失测试

输出格式：
- 只输出Python代码
- 不要包含markdown的代码块标记
- 不要有其他解释或说明
```

### 提示词构造

```python
def _build_prompt(
        self,
        request: TestCaseGenerationRequest,
        few_shot_examples: List[Dict],
        rag_cases: List[Dict]
    ) -> str:
        """构造完整的用户提示词"""
        
        prompt = ""
        
        # 第1部分：Few-shot示例
        prompt += "【参考示例】\n\n"
        for i, example in enumerate(few_shot_examples, 1):
            prompt += f"示例{i}：{example['name']}\n"
            prompt += f"接口: {example['method']} {example['url']}\n"
            prompt += f"代码:\n{example['test_code']}\n\n"
        
        # 第2部分：RAG检索的相似用例
        if rag_cases:
            prompt += "【项目相似用例】\n\n"
            for i, case in enumerate(rag_cases, 1):
                prompt += f"参考用例{i}（相似度：{case['score']:.1%}）\n"
                prompt += f"接口: {case['interface_name']}\n"
                prompt += f"代码:\n{case['test_code'][:500]}...\n\n"
        
        # 第3部分：当前接口信息
        prompt += "【要生成的接口】\n\n"
        prompt += f"接口名称: {request.interface_name}\n"
        prompt += f"HTTP方法: {request.http_method}\n"
        prompt += f"URL: {request.url}\n"
        prompt += f"描述: {request.description}\n"
        
        # 请求参数
        if request.request_params:
            prompt += f"\n请求参数:\n"
            prompt += json.dumps(request.request_params, indent=2, ensure_ascii=False)
        
        # 响应格式
        if request.response_schema:
            prompt += f"\n响应格式:\n"
            prompt += json.dumps(request.response_schema, indent=2, ensure_ascii=False)
        
        # 依赖
        if request.dependencies:
            prompt += f"\n依赖的前置接口: {', '.join(request.dependencies)}\n"
        
        # 认证需求
        if request.requires_auth:
            prompt += "\n说明: 该接口需要在请求头中提供 Authorization: Bearer {token}\n"
        
        prompt += "\n请为上述接口生成完整的Pytest测试用例。\n"
        
        return prompt
```

## Few-shot管理

### 动态Few-shot构造

```python
class FewShotManager:
    """管理Few-shot示例"""
    
    def __init__(self, db_session):
        self.db = db_session
    
    async def get_best_examples(
        self,
        interface_info: Dict,
        method: str,
        count: int = 3
    ) -> List[Dict]:
        """
        从数据库获取最好的Few-shot示例
        """
        
        # 查询条件：HTTP方法相同
        query = self.db.query(TestCase).filter(
            TestCase.http_method == interface_info['method']
        )
        
        # 排序：优先使用最新的、质量最高的
        cases = query.order_by(
            TestCase.quality_score.desc(),
            TestCase.created_at.desc()
        ).limit(count).all()
        
        return [
            {
                'name': case.interface_name,
                'method': case.http_method,
                'url': case.url,
                'test_code': case.test_code
            }
            for case in cases
        ]
    
    async def update_quality_score(self, test_case_id: str, score: float):
        """
        更新测试用例的质量分数
        （用于选择更好的Few-shot示例）
        """
        case = await self.db.query(TestCase).filter(
            TestCase.id == test_case_id
        ).first()
        
        if case:
            # 使用指数平滑法更新分数
            case.quality_score = 0.7 * case.quality_score + 0.3 * score
            await self.db.commit()
```

### RAG检索实现

```python
async def _retrieve_similar_cases(
        self,
        request: TestCaseGenerationRequest
    ) -> List[Dict]:
        """
        从RAG向量库检索相似的测试用例
        """
        
        try:
            # 查询条件：项目相同
            query_interface = {
                'name': request.interface_name,
                'method': request.http_method,
                'url': request.url,
                'description': request.description
            }
            
            # 从向量存储检索
            results = self.vector_store.retrieve_similar_cases(
                query_interface,
                project_id=request.project_id,
                top_k=self.rag_top_k
            )
            
            # 过滤低相似度的结果
            filtered = [
                r for r in results
                if r.get('score', 0) > 0.5
            ]
            
            logger.info(f"📚 检索到 {len(filtered)} 个相似用例")
            
            return filtered
            
        except Exception as e:
            logger.warning(f"⚠️ RAG检索失败: {str(e)}，继续使用Few-shot")
            return []

async def _store_to_rag(
        self,
        request: TestCaseGenerationRequest,
        test_code: str
    ):
        """
        将新生成的测试用例存储到RAG向量库
        （用于后续的RAG检索）
        """
        
        try:
            self.vector_store.add_test_case(
                case_id=f"test_{request.interface_id}_{datetime.now().timestamp()}",
                interface_info={
                    'name': request.interface_name,
                    'method': request.http_method,
                    'url': request.url,
                    'description': request.description
                },
                test_code=test_code,
                project_id=request.project_id,
                metadata={
                    'generated_at': datetime.now().isoformat(),
                    'quality_score': 0.8  # 新生成的用例初始分数
                }
            )
            
            logger.info(f"💾 用例已存储到RAG库")
            
        except Exception as e:
            logger.warning(f"⚠️ RAG存储失败: {str(e)}")
```

## 错误处理和重试

### 重试机制

```python
from tenacity import retry, stop_after_attempt, wait_exponential

class RobustTestCaseGenerator(TestCaseGenerator):
    """具有健壮错误处理的生成器"""
    
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10)
    )
    async def generate_with_retry(
        self,
        request: TestCaseGenerationRequest
    ) -> TestCaseGenerationResult:
        """带重试的生成方法"""
        return await self.generate(request)
    
    async def generate_with_fallback(
        self,
        request: TestCaseGenerationRequest
    ) -> TestCaseGenerationResult:
        """
        生成测试用例，支持降级方案
        """
        
        try:
            # 尝试完整流程：Few-shot + RAG + LLM
            return await self.generate(request)
            
        except Exception as e1:
            logger.warning(f"⚠️ 完整流程失败: {str(e1)}")
            
            try:
                # 降级1：只用Few-shot，不用RAG
                return await self._generate_with_few_shot_only(request)
                
            except Exception as e2:
                logger.warning(f"⚠️ Few-shot流程失败: {str(e2)}")
                
                # 降级2：使用模板生成基础用例
                return self._generate_from_template(request)

    def _generate_from_template(
        self,
        request: TestCaseGenerationRequest
    ) -> TestCaseGenerationResult:
        """使用模板生成基础测试用例"""
        
        template = f'''
import pytest
import requests

BASE_URL = "{{BASE_URL}}"

def test_{request.interface_name.lower()}():
    """测试 {request.interface_name}"""
    
    response = requests.{request.http_method.lower()}(
        f"{{BASE_URL}}{request.url}"
    )
    
    assert response.status_code in [200, 201, 204]
    
    if response.status_code != 204:
        data = response.json()
        assert "code" in data or "data" in data
'''
        
        return TestCaseGenerationResult(
            interface_id=request.interface_id,
            test_code=template,
            test_cases=[{
                'name': f'test_{request.interface_name.lower()}',
                'type': 'basic'
            }],
            generated_at=datetime.now(),
            model_used='template',
            tokens_used=0,
            few_shot_count=0,
            rag_score=None
        )
```

## 性能优化建议

1. **缓存**：相同接口的Few-shot示例可以缓存
2. **并发**：多个接口可以并发生成
3. **预热**：启动时预加载embedding模型
4. **批处理**：可以批量生成多个接口的用例

```python
# 示例：并发生成
async def generate_multiple(
    generator: TestCaseGenerator,
    requests: List[TestCaseGenerationRequest]
) -> List[TestCaseGenerationResult]:
    """并发生成多个接口的测试用例"""
    
    import asyncio
    
    tasks = [
        generator.generate(req)
        for req in requests
    ]
    
    return await asyncio.gather(*tasks)
```
