## 一、知识库技术栈

```
ChromaDB (向量数据库)
    ↓
通义千问 text-embedding-v3 (向量化)
    ↓
BM25算法 (关键词检索)
    ↓
通义千问 qwen3-rerank (重排序)
    ↓
Neo4j (知识图谱，可选)
```

#### 向量化（Embedding）
```python
def embed(self, texts: List[str], use_threading: bool = True) -> List[List[float]]:
    """使用通义千问API生成向量，支持多线程并发"""
    from dashscope import TextEmbedding
    
    resp = TextEmbedding.call(
        model="text-embedding-v3",
        input=texts  # 批量处理
    )
    
    embeddings = [item["embedding"] for item in resp["output"]["embeddings"]]
    return embeddings
```

#### 存储文档
```python
async def add_documents(
    self,
    document_id: int,
    chunks: List[str],
    metadata_list: Optional[List[Dict]] = None
):
    """添加文档到ChromaDB"""
    # 1. 生成向量
    vectors = self.embed(chunks, use_threading=True)
    
    # 2. 准备数据
    ids = [f"{document_id}_{i}" for i in range(len(chunks))]
    metadatas = [{"document_id": str(document_id), **meta} for meta in metadata_list]
    
    # 3. 存储到ChromaDB
    self.collection.add(
        ids=ids,
        embeddings=vectors,
        documents=chunks,
        metadatas=metadatas
    )
```

#### 混合检索（核心）
```python
async def search(
    self,
    query: str,
    top_k: int = 10,
    use_rerank: bool = True
) -> List[Dict]:
    """混合检索：向量检索 + BM25 + 重排序"""
    
    # 1. 向量检索
    query_vector = self.embed([query])[0]
    results = self.collection.query(
        query_embeddings=[query_vector],
        n_results=top_k * 2
    )
    
    # 2. BM25关键词检索
    from rank_bm25 import BM25Okapi
    corpus = [doc for doc in results["documents"][0]]
    bm25 = BM25Okapi([doc.split() for doc in corpus])
    bm25_scores = bm25.get_scores(query.split())
    
    # 3. 分数融合（向量70% + BM25 30%）
    for i, result in enumerate(vector_results):
        result["final_score"] = result["score"] * 0.7 + bm25_scores[i] * 0.3
    
    # 4. 重排序（可选）
    if use_rerank:
        reranker = RerankerService()
        reranked = reranker.rerank(query, [r["chunk_text"] for r in vector_results])
        # 更新排序...
    
    return vector_results[:top_k]
```

### 2.2 RerankerService - 重排序服务

**文件**: `backend/app/services/reranker_service.py`

```python
class RerankerService:
    def rerank(self, query: str, texts: List[str]) -> List[Tuple[int, float]]:
        """使用通义千问qwen3-rerank API重排序"""
        from dashscope import TextReRank
        
        resp = TextReRank.call(
            model="qwen3-rerank",
            query=query,
            documents=texts
        )
        
        # 返回 [(索引, 分数), ...]
        return [(item["index"], item["relevance_score"]) 
                for item in resp["output"]["results"]]
```

### 2.3 HybridRAGService - 混合RAG服务

**文件**: `backend/app/services/rag_service.py`

```python
class HybridRAGService:
    def __init__(self):
        self.vector_service = VectorService()
        self.reranker_service = RerankerService()
    
    async def hybrid_search(
        self,
        query: str,
        project_id: int,
        top_k: int = 10
    ) -> List[Dict]:
        """混合检索入口"""
        results = await self.vector_service.search(query, top_k, use_rerank=True)
        return results
    
    async def graph_rag_search(
        self,
        query: str,
        project_id: int,
        top_k: int = 10
    ) -> Dict:
        """GraphRAG：图谱 + 向量检索"""
        # 1. Neo4j图谱检索
        graph_results = self.db_service.query_knowledge_graph(...)
        
        # 2. 向量检索
        vector_results = await self.hybrid_search(query, project_id, top_k)
        
        # 3. 融合结果
        return {
            "graph_context": self._extract_graph_context(graph_results),
            "vector_context": "\n".join([r["text"] for r in vector_results])
        }
```

## 三、使用示例

### 3.1 索引文档

```python
# 文档解析完成后，索引到知识库
vector_service = VectorService()

# 方式1: 添加分类内容（推荐）

    document_id=document.id,
    parsed_data={
        "text": ["文本块1", "文本块2"],
        "images": ["图片描述1"],
        "tables": [{"headers": [...], "rows": [...]}],
        "formulas": ["公式1"]
    }
)

# 方式2: 直接添加文档块

    document_id=document.id,
    chunks=["文档块1", "文档块2"],
    metadata_list=[{"source": "text"}, {"source": "table"}]
)
```

```python

    project_id=1,

# 方式3: GraphRAG检索（图谱增强）

    query="登录接口的依赖关系",
    project_id=1,
    top_k=10
)
```

## 七、性能优化要点

1. **多线程并发**: Embedding生成使用线程池（最多5个并发）
2. **批量处理**: 每批10条文本，减少API调用次数
3. **异步处理**: 使用asyncio避免阻塞
4. **Redis缓存**: 缓存检索结果，提高响应速度
5. **持久化存储**: ChromaDB数据持久化，重启不丢失
